var Alexa = require('alexa-sdk')

exports.handler = function (event, context, callback) {
  var alexa = Alexa.handler(event, context)
  alexa.registerHandlers(handlers)
  alexa.execute()
}

var handlers = {
  'LaunchRequest': function () {
    this.emit(':ask', 'Welcome to the sample skill. Say help for help.')
  },
  
  'Intent1': function () {
    this.emit(':ask', 'This is the first intent.')
  },
  
  'Intent2': function () {
    var input1 = this.event.request.intent.slots.InputOne.value.toString()
    var input2 = this.event.request.intent.slots.InputTwo.value.toString()
    var response = doSomething(input1, input2)
    this.emit(':ask', response)
  },

  'AMAZON.HelpIntent': function () {
    this.emit(':ask', 'Insert some help message here.')
  },

  'AMAZON.StopIntent': function () {
    this.emit(':tell', 'Have a good rest of your day!')
  }
}

function doSomething (input1, input2) {
  if (input1.localeCompare(input2)) {
    return "They are the same string!"
  } else {
    return "They are different!"
  }
}

Deployment #

Alexa Skill Kit #

Go to this page and select ‚ÄòCreate Skill‚Äô. In there you should see options to ‚Äòadd intents‚Äô and ‚Äòcustom slot types‚Äô. You should add the appropriate slot types and intents accordingly from above. After you are done with that, select ‚ÄòEndpoint‚Äô from the side bar and select ‚ÄòAWS Lambda ARN‚Äô as the skill‚Äôs service endpoint. Make a copy of the Skill ID as you will need it later.

AWS Lambda #

Create an AWS Lambda function selecting ‚Äòfrom scratch‚Äô. After that, you should have an option to ‚Äòadd trigger‚Äô where you would select ‚ÄòAlexa Skills Kit‚Äô. Now in the configuration dialog, you should put in the Skill ID you took a copy of previously then select ‚ÄòAdd‚Äô.

After this, select back to the box that labeled your project name. There should be a ‚ÄòFunction code‚Äô dialog box appeared below. While you can definitely copy and paste your code onto the online text editor, there might be chance where the packages you have installed and used in your code is included by default. It is also possible that you have more than 1 JavaScript files for the skill.

In this case, you would select ‚ÄòUpload a .ZIP file‚Äô under ‚ÄòCode entry type‚Äô. Locally, zip up all the JavaScript files along with the node_modules folder then upload the resulting zip file.

Submission #

When you are done, you can go back to the Alexa Skill Kit page, select the ‚ÄòDistribution‚Äô tab and fill out the appropriate information before submitting your skill for review. Hopefully everything goes well and your skill gets published at the Alexa Skill Store!
